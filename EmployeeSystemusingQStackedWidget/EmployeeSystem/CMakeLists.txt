cmake_minimum_required(VERSION 3.19)
project(EmployeeSystem LANGUAGES CXX)

find_package(Qt6 6.5 REQUIRED COMPONENTS Core  Sql Widgets)

qt_standard_project_setup()

qt_add_executable(EmployeeSystem
    WIN32 MACOSX_BUNDLE
    main.cpp
    mainwindow.cpp
    mainwindow.h
    mainwindow.ui
    employeedashboard.cpp
    employeedashboard.h
    employeedashboard.ui
    employeelogin.cpp
    employeelogin.h
    employeelogin.ui
)

target_link_libraries(EmployeeSystem
    PRIVATE
        Qt::Core
        Qt::Widgets
        Qt::Sql
)

include(GNUInstallDirs)

install(TARGETS EmployeeSystem
    BUNDLE  DESTINATION .
    RUNTIME DESTINATION ${CMAKE_INSTALL_BINDIR}
    LIBRARY DESTINATION ${CMAKE_INSTALL_LIBDIR}
)

qt_generate_deploy_app_script(
    TARGET EmployeeSystem
    OUTPUT_SCRIPT deploy_script
    NO_UNSUPPORTED_PLATFORM_ERROR
)
install(SCRIPT ${deploy_script})
